#1 count how many lines there are in each file
wc -l ../data/fasta/*.fasta

#2 print everything starting from the second line for E. coli
tail +2 ../data/fasta/E.coli.fasta

#3 count the sequence length of this genome
tail +2 ../data/fasta/E.coli.fasta | tr -d '\t' | wc -m

#4 count the matches of a particular sequence "ATGC" in the genome of E. coli (hint: Start by removing the first line and removing newline characters)
tail +2 ../data/fasta/E.coli.fasta | tr -s '\t' | grep -o -i "ATGC" | wc -l

#5 compute the AT/GC ratio. That is, the (A+T)/(G+C) ratio. This is a summary measure of base composition of double-stranded DNA. DNA from different organisms and lineages has different ratios of the A-to-T and G-to-C base pairs. For example DNA from organisms that live in hot springs have a higher GC content, which takes advantage of the increased thermal stability of the GC base pair (google “Chargaff’s rule”).
echo "$(tail +2 ../data/fasta/E.coli.fasta | grep -o 'A\|T' | wc -l ) / $(tail +2 ../data/fasta/E.coli.fasta | grep -o 'G\|C' | wc -l )" | bc -l
